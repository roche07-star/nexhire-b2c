# 개인정보 처리 구현 방안 (Updated)

## 3.2 Eve (B2B) - 제3자 제공 동의 (오프라인)

### 동의 프로세스

```
1. 헤드헌터가 후보자에게 직접 연락 (전화/이메일/대면)
   └─ 채용사 정보, 포지션, 제공 항목 설명
   └─ 구두 동의 또는 서면 동의서 수령

2. Eve 시스템에 동의 기록 입력
   └─ 동의 받은 일시, 방법, 동의서 파일 업로드

3. 후보자 정보 제공
   └─ 채용사에 이력서 전달
   └─ 제공 기록 저장 (3년 보관)
```

### 구현 코드

#### A. 동의 기록 등록 API

```typescript
// eve/api/consents/record/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { auth } from '@/auth'
import { supabase } from '@/lib/supabase'

export async function POST(req: NextRequest) {
  try {
    const session = await auth()
    if (!session?.user?.email) {
      return NextResponse.json({ error: '로그인이 필요합니다.' }, { status: 401 })
    }

    // B2B 헤드헌터만 접근 가능
    const { data: user } = await supabase
      .from('users')
      .select('service_type, user_type')
      .eq('email', session.user.email)
      .single()

    if (user?.service_type !== 'B2B') {
      return NextResponse.json({ error: '권한이 없습니다.' }, { status: 403 })
    }

    const {
      candidateEmail,
      candidateName,
      candidatePhone,
      recipientCompany,
      position,
      providedItems,
      consentMethod, // 'phone', 'email', 'in-person', 'document'
      consentDocumentUrl, // 동의서 파일 URL (선택)
      agreedAt,
      notes
    } = await req.json()

    // 1. 후보자 존재 확인 (선택 - Adam B2C 회원일 수도, 아닐 수도 있음)
    const { data: candidate } = await supabase
      .from('users')
      .select('email')
      .eq('email', candidateEmail)
      .single()

    // 2. 동의 기록 저장
    const { data: consent, error } = await supabase
      .from('consents')
      .insert({
        user_email: candidateEmail,
        consent_type: 'third_party_provision',
        consent_version: 'v1.0.0',
        is_agreed: true,
        agreed_at: new Date(agreedAt),
        
        // 제3자 제공 정보
        recipient_company: recipientCompany,
        provision_purpose: `${recipientCompany} ${position} 채용 전형`,
        provided_items: providedItems,
        provision_period: '채용 전형 종료 시까지',
        
        // 오프라인 동의 관련 정보
        consent_method: consentMethod,
        consent_document_url: consentDocumentUrl,
        recorded_by: session.user.email,
        notes: notes
      })
      .select()
      .single()

    if (error) {
      console.error('[consents/record]', error)
      return NextResponse.json({ error: '동의 기록 저장 실패' }, { status: 500 })
    }

    // 3. 후보자 정보 등록 (Adam 회원이 아닌 경우)
    if (!candidate) {
      await supabase.from('external_candidates').insert({
        email: candidateEmail,
        name: candidateName,
        phone: candidatePhone,
        source: 'offline',
        registered_by: session.user.email,
        created_at: new Date()
      })
    }

    // 4. 감사 로그
    await supabase.from('audit_logs').insert({
      action: 'record_consent',
      actor_email: session.user.email,
      target_email: candidateEmail,
      details: {
        recipient_company: recipientCompany,
        position: position,
        consent_method: consentMethod
      },
      ip_address: req.headers.get('x-forwarded-for') || req.headers.get('x-real-ip'),
      created_at: new Date()
    })

    return NextResponse.json({ 
      success: true, 
      consentId: consent.id 
    })

  } catch (e) {
    console.error('[consents/record]', e)
    return NextResponse.json({ error: '서버 오류' }, { status: 500 })
  }
}
```

#### B. 후보자 정보 제공 API

```typescript
// eve/api/candidates/provide/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { auth } from '@/auth'
import { supabase } from '@/lib/supabase'

export async function POST(req: NextRequest) {
  try {
    const session = await auth()
    if (!session?.user?.email) {
      return NextResponse.json({ error: '로그인이 필요합니다.' }, { status: 401 })
    }

    const {
      candidateEmail,
      recipientCompany,
      position,
      providedItems,
      resumeFileUrl // 이력서 파일 URL
    } = await req.json()

    // 1. 동의 확인
    const { data: consent } = await supabase
      .from('consents')
      .select('*')
      .eq('user_email', candidateEmail)
      .eq('consent_type', 'third_party_provision')
      .eq('recipient_company', recipientCompany)
      .eq('is_agreed', true)
      .is('withdrawn_at', null)
      .order('agreed_at', { ascending: false })
      .limit(1)
      .single()

    if (!consent) {
      return NextResponse.json({ 
        error: '동의 기록이 없습니다. 먼저 동의를 받고 기록해주세요.' 
      }, { status: 400 })
    }

    // 2. 제공 기록 저장 (3년 보관)
    const retentionUntil = new Date()
    retentionUntil.setFullYear(retentionUntil.getFullYear() + 3)

    const { data: provisionLog, error } = await supabase
      .from('third_party_provision_logs')
      .insert({
        candidate_email: candidateEmail,
        recipient_company: recipientCompany,
        position: position,
        provided_items: providedItems,
        resume_file_url: resumeFileUrl,
        provided_by: session.user.email,
        provided_at: new Date(),
        retention_until: retentionUntil,
        consent_id: consent.id
      })
      .select()
      .single()

    if (error) {
      console.error('[candidates/provide]', error)
      return NextResponse.json({ error: '제공 기록 저장 실패' }, { status: 500 })
    }

    // 3. 감사 로그
    await supabase.from('audit_logs').insert({
      action: 'provide_candidate_info',
      actor_email: session.user.email,
      target_email: candidateEmail,
      details: {
        recipient_company: recipientCompany,
        position: position,
        provided_items: providedItems
      },
      ip_address: req.headers.get('x-forwarded-for') || req.headers.get('x-real-ip'),
      created_at: new Date()
    })

    // 4. 채용사에 이메일 발송 (선택)
    // await sendCandidateInfoEmail({
    //   to: recipientCompanyEmail,
    //   candidateEmail,
    //   resumeUrl: resumeFileUrl
    // })

    return NextResponse.json({ 
      success: true,
      provisionId: provisionLog.id
    })

  } catch (e) {
    console.error('[candidates/provide]', e)
    return NextResponse.json({ error: '서버 오류' }, { status: 500 })
  }
}
```

```tsx
// eve/components/ConsentRecordForm.tsx
'use client'

import { useState } from 'react'

export default function ConsentRecordForm({ candidateEmail, onSuccess }: {
  candidateEmail: string
  onSuccess: () => void
}) {
  const [formData, setFormData] = useState({
    candidateName: '',
    candidatePhone: '',
    recipientCompany: '',
    position: '',
    providedItems: ['이름', '전화번호', '이메일', '이력서'],
    consentMethod: 'phone' as 'phone' | 'email' | 'in-person' | 'document',
    agreedAt: new Date().toISOString().slice(0, 16),
    notes: ''
  })
  const [file, setFile] = useState<File | null>(null)
  const [loading, setLoading] = useState(false)

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    setLoading(true)

    try {
      // 1. 동의서 파일 업로드 (있는 경우)
      let consentDocumentUrl = null
      if (file) {
        const formData = new FormData()
        formData.append('file', file)
        const uploadRes = await fetch('/api/upload/consent-document', {
          method: 'POST',
          body: formData
        })
        const uploadData = await uploadRes.json()
        consentDocumentUrl = uploadData.url
      }

      // 2. 동의 기록 등록
      const res = await fetch('/api/consents/record', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          ...formData,
          candidateEmail,
          consentDocumentUrl
        })
      })

      if (!res.ok) throw new Error('동의 기록 등록 실패')

      alert('동의 기록이 등록되었습니다.')
      onSuccess()

    } catch (err) {
      console.error(err)
      alert('동의 기록 등록에 실패했습니다.')
    } finally {
      setLoading(false)
    }
  }

  return (
    <form onSubmit={handleSubmit} className="consent-record-form">
      <h3>제3자 제공 동의 기록</h3>

      <div className="form-section">
        <h4>후보자 정보</h4>
        <input
          type="email"
          value={candidateEmail}
          disabled
          className="form-input"
        />
        <input
          type="text"
          placeholder="후보자 이름"
          value={formData.candidateName}
          onChange={(e) => setFormData({ ...formData, candidateName: e.target.value })}
          required
          className="form-input"
        />
        <input
          type="tel"
          placeholder="후보자 전화번호"
          value={formData.candidatePhone}
          onChange={(e) => setFormData({ ...formData, candidatePhone: e.target.value })}
          required
          className="form-input"
        />
      </div>

      <div className="form-section">
        <h4>채용사 정보</h4>
        <input
          type="text"
          placeholder="채용사명 (예: 삼성전자)"
          value={formData.recipientCompany}
          onChange={(e) => setFormData({ ...formData, recipientCompany: e.target.value })}
          required
          className="form-input"
        />
        <input
          type="text"
          placeholder="포지션 (예: 마케팅 매니저)"
          value={formData.position}
          onChange={(e) => setFormData({ ...formData, position: e.target.value })}
          required
          className="form-input"
        />
      </div>

      <div className="form-section">
        <h4>제공 항목</h4>
        <div className="checkbox-group">
          {['이름', '전화번호', '이메일', '이력서', '경력기술서'].map(item => (
            <label key={item}>
              <input
                type="checkbox"
                checked={formData.providedItems.includes(item)}
                onChange={(e) => {
                  if (e.target.checked) {
                    setFormData({ ...formData, providedItems: [...formData.providedItems, item] })
                  } else {
                    setFormData({ 
                      ...formData, 
                      providedItems: formData.providedItems.filter(i => i !== item) 
                    })
                  }
                }}
              />
              {item}
            </label>
          ))}
        </div>
      </div>

      <div className="form-section">
        <h4>동의 받은 방법</h4>
        <select
          value={formData.consentMethod}
          onChange={(e) => setFormData({ 
            ...formData, 
            consentMethod: e.target.value as any 
          })}
          className="form-select"
        >
          <option value="phone">전화 (구두 동의)</option>
          <option value="email">이메일</option>
          <option value="in-person">대면</option>
          <option value="document">서면 동의서</option>
        </select>
      </div>

      <div className="form-section">
        <h4>동의 받은 일시</h4>
        <input
          type="datetime-local"
          value={formData.agreedAt}
          onChange={(e) => setFormData({ ...formData, agreedAt: e.target.value })}
          required
          className="form-input"
        />
      </div>

      {formData.consentMethod === 'document' && (
        <div className="form-section">
          <h4>동의서 파일 업로드</h4>
          <input
            type="file"
            accept=".pdf,.jpg,.jpeg,.png"
            onChange={(e) => setFile(e.target.files?.[0] || null)}
            className="form-file"
          />
          <p className="form-help">PDF, JPG, PNG 파일만 업로드 가능</p>
        </div>
      )}

      <div className="form-section">
        <h4>메모 (선택)</h4>
        <textarea
          placeholder="예) 전화 통화 후 구두 동의 받음, 동의 내용 상세 설명 완료"
          value={formData.notes}
          onChange={(e) => setFormData({ ...formData, notes: e.target.value })}
          className="form-textarea"
          rows={3}
        />
      </div>

      <div className="form-notice">
        <h4>⚠️ 유의사항</h4>
        <ul>
          <li>후보자에게 채용사명, 포지션, 제공 항목을 명확히 안내했는지 확인하세요.</li>
          <li>동의 기록은 법적 효력이 있으므로 정확하게 입력하세요.</li>
          <li>제공 기록은 3년간 보관됩니다.</li>
        </ul>
      </div>

      <button 
        type="submit" 
        disabled={loading}
        className="form-submit"
      >
        {loading ? '등록 중...' : '동의 기록 등록'}
      </button>
    </form>
  )
}
```

#### B. 후보자 제공 확인

```tsx
// eve/components/CandidateProvisionConfirm.tsx
'use client'

export default function CandidateProvisionConfirm({ 
  candidate, 
  company, 
  position,
  onConfirm 
}: {
  candidate: { email: string; name: string; resumeUrl: string }
  company: string
  position: string
  onConfirm: () => void
}) {
  const [loading, setLoading] = useState(false)

  async function handleProvide() {
    if (!confirm(`${candidate.name}님의 정보를 ${company}에 제공하시겠습니까?`)) {
      return
    }

    setLoading(true)
    try {
      const res = await fetch('/api/candidates/provide', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          candidateEmail: candidate.email,
          recipientCompany: company,
          position: position,
          providedItems: ['이름', '전화번호', '이메일', '이력서'],
          resumeFileUrl: candidate.resumeUrl
        })
      })

      if (!res.ok) {
        const data = await res.json()
        throw new Error(data.error || '제공 실패')
      }

      alert('후보자 정보가 제공되었습니다.')
      onConfirm()

    } catch (err: any) {
      console.error(err)
      alert(err.message || '정보 제공에 실패했습니다.')
    } finally {
      setLoading(false)
    }
  }

  return (
    <div className="provision-confirm">
      <h3>후보자 정보 제공</h3>
      
      <div className="provision-info">
        <div className="info-row">
          <span className="label">후보자:</span>
          <span className="value">{candidate.name} ({candidate.email})</span>
        </div>
        <div className="info-row">
          <span className="label">채용사:</span>
          <span className="value">{company}</span>
        </div>
        <div className="info-row">
          <span className="label">포지션:</span>
          <span className="value">{position}</span>
        </div>
      </div>

      <div className="provision-notice">
        <p>✅ 동의 기록이 확인되었습니다.</p>
        <p>📋 제공 기록은 3년간 보관됩니다.</p>
      </div>

      <button 
        onClick={handleProvide}
        disabled={loading}
        className="btn-provide"
      >
        {loading ? '제공 중...' : '정보 제공하기'}
      </button>
    </div>
  )
}
```

### 데이터베이스 스키마 업데이트

```sql
-- consents 테이블에 오프라인 동의 관련 컬럼 추가
ALTER TABLE consents ADD COLUMN IF NOT EXISTS consent_method TEXT; -- 'phone', 'email', 'in-person', 'document'
ALTER TABLE consents ADD COLUMN IF NOT EXISTS consent_document_url TEXT; -- 동의서 파일 URL
ALTER TABLE consents ADD COLUMN IF NOT EXISTS recorded_by TEXT; -- 기록한 헤드헌터 이메일
ALTER TABLE consents ADD COLUMN IF NOT EXISTS notes TEXT; -- 메모

-- external_candidates 테이블 (Adam 회원이 아닌 후보자)
CREATE TABLE IF NOT EXISTS external_candidates (
  email TEXT PRIMARY KEY,
  name TEXT NOT NULL,
  phone TEXT,
  source TEXT DEFAULT 'offline', -- 'offline', 'referral', etc.
  registered_by TEXT NOT NULL, -- 등록한 헤드헌터
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- third_party_provision_logs 테이블 업데이트
CREATE TABLE IF NOT EXISTS third_party_provision_logs (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  candidate_email TEXT NOT NULL,
  recipient_company TEXT NOT NULL,
  position TEXT,
  provided_items TEXT[] NOT NULL,
  resume_file_url TEXT,
  provided_by TEXT NOT NULL, -- 제공한 헤드헌터
  provided_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  retention_until TIMESTAMP WITH TIME ZONE NOT NULL, -- 3년 후
  consent_id UUID REFERENCES consents(id),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_provision_logs_candidate ON third_party_provision_logs(candidate_email);
CREATE INDEX idx_provision_logs_company ON third_party_provision_logs(recipient_company);
CREATE INDEX idx_provision_logs_retention ON third_party_provision_logs(retention_until);
```
